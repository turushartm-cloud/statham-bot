// ── v2.1: Поиск манипуляционных свечей в lookback окне ──────────
// Манипуляционная свеча = свеча с длинным фитилём (стоп-хант, вынос стопов)
// Нижний фитиль > N*ATR → маркетмейкер выбивал лонг-стопы
// Верхний фитиль > N*ATR → маркетмейкер выбивал шорт-стопы
f_find_manip_candles(int _lb, float _wick_mult) =>
    float _ml = na
    float _mh = na
    for i = 0 to _lb - 1
        float _body_low  = math.min(open[i], close[i])
        float _body_high = math.max(open[i], close[i])
        float _lw = _body_low  - low[i]
        float _uw = high[i]    - _body_high
        if _lw > atr_val * _wick_mult
            _ml := na(_ml) ? low[i] : math.min(_ml, low[i])
        if _uw > atr_val * _wick_mult
            _mh := na(_mh) ? high[i] : math.max(_mh, high[i])
    [_ml, _mh]
[manip_low_candle, manip_high_candle] = f_find_manip_candles(struct_sl_lookback, manip_wick_mult)
f_smart_sl(bool is_long, float entry, float atr_fallback) =>
    // ── v2 UPGRADE: Structure-Based SL (защита от стоп-ханта) ─────────
    // Теперь SL ставится ЗА ближайший структурный свинг-лой/хай (N свечей)
    // + буфер % от цены. Минимальная дистанция = sl_min_atr_mult × ATR
    // ───────────
    // ── v2 FIX: extract ta.lowest/ta.highest outside conditional branches ──
    float _sw_low_all  = ta.lowest(low,  struct_sl_lookback)
    float _sw_high_all = ta.highest(high, struct_sl_lookback)
    float sl = na
    array<float> sl_candidates = array.new_float(0)
    // ── Эффективный буфер за OB/FVG (больше, чем раньше) ──────────
    // ✅ FIX: nz() guard — atr_val может быть na на первых барах нового листинга
    float _safe_atr   = nz(atr_val, entry * 0.015)
    float eff_ob_buf  = _safe_atr * ob_sl_buffer_atr
    float eff_fvg_buf = _safe_atr * (ob_sl_buffer_atr * 0.7)
    // ── Минимальная дистанция SL от входа (адаптивная по ТФ) ────────────
    float min_sl_dist = _safe_atr * sl_min_atr_eff
    if is_long
        // ── 1. Структурный свинг-лой за последние N свечей (главный кандидат) ─
        if use_struct_sl
            float sw_low_n = _sw_low_all
            float struct_candidate = sw_low_n * (1.0 - sl_buffer_pct / 100.0)
            array.push(sl_candidates, struct_candidate)
        // ── 2. Манипуляционная свеча (длинный нижний фитиль) — SL за её лой ──
        // Это и есть истинный уровень стоп-ханта: ММ выбил стопы ниже, потом вернулся
        if use_manip_candle and not na(manip_low_candle) and manip_low_candle < entry
            array.push(sl_candidates, manip_low_candle - eff_ob_buf * 0.5)
        // ── 3. Лой свечи sweep_low — за ней и буфер ─────────
        if sweep_low
            array.push(sl_candidates, math.min(low[1], low[2]) - eff_ob_buf)
        // ── 4. За нижнюю границу FVG бычьего ───────────
        if not na(last_fvg_bull_bottom) and last_fvg_bull_bottom < entry
            array.push(sl_candidates, last_fvg_bull_bottom - eff_fvg_buf)
        // ── 4. За нижнюю границу Order Block бычьего ───────────
        if not na(nearest_ob_bull) and nearest_ob_bull < entry
            array.push(sl_candidates, nearest_ob_bull - eff_ob_buf)
        // ── 5. За Fib 38.2% ───────────
        if not na(af_382) and af_382 < entry
            array.push(sl_candidates, af_382 - atr_val * 0.4)
        // ── 6. За нижнюю границу HVN зоны ──────────
        if not na(hvn1_price) and hvn1_price < entry
            array.push(sl_candidates, hvn1_price - hvn_zone_size - atr_val * 0.2)
        // ── Выбор лучшего SL кандидата ─────────
        if array.size(sl_candidates) > 0
            float best_sl = na
            float best_distance = na
            for i = 0 to array.size(sl_candidates) - 1
                float candidate = array.get(sl_candidates, i)
                float distance  = entry - candidate
                // Увеличен порог "слишком близко": было 0.5 ATR, теперь min_sl_dist
                bool not_too_close = distance >= min_sl_dist
                bool not_too_far   = distance <= atr_fallback * 4
                if not_too_close and not_too_far
                    // Выбираем кандидата ближайшего к входу (но за структурой)
                    if na(best_sl) or distance < best_distance
                        best_sl       := candidate
                        best_distance := distance
            if na(best_sl)
                // Все кандидаты слишком близко — расширяем за структурный свинг + мин. дистанция
                float sw_low_fb = use_struct_sl ? _sw_low_all : current_swing_low
                float buf = sw_low_fb * (sl_buffer_pct / 100.0)
                float raw_sl = sw_low_fb - buf
                sl := (entry - raw_sl) >= min_sl_dist ? raw_sl : entry - min_sl_dist
            else
                sl := best_sl
        else
            float sw_low_fb = use_struct_sl ? _sw_low_all : current_swing_low
            float buf = sw_low_fb * (sl_buffer_pct / 100.0)
            float raw_sl = sw_low_fb - buf
            sl := (entry - raw_sl) >= min_sl_dist ? raw_sl : entry - min_sl_dist
    else
        // ── 1. Структурный свинг-хай за последние N свечей (главный кандидат) ─
        if use_struct_sl
            float sw_high_n = _sw_high_all
            float struct_candidate = sw_high_n * (1.0 + sl_buffer_pct / 100.0)
            array.push(sl_candidates, struct_candidate)
        // ── 2. Манипуляционная свеча (длинный верхний фитиль) — SL за её хай ─
        // ММ выбил шорт-стопы выше, потом вернулся — это истинный уровень выноса
        if use_manip_candle and not na(manip_high_candle) and manip_high_candle > entry
            array.push(sl_candidates, manip_high_candle + eff_ob_buf * 0.5)
        // ── 3. Хай свечи sweep_high ────────────
        if sweep_high
            array.push(sl_candidates, math.max(high[1], high[2]) + eff_ob_buf)
        // ── 4. За верхнюю границу FVG медвежьего ──────────
        if not na(last_fvg_bear_top) and last_fvg_bear_top > entry
            array.push(sl_candidates, last_fvg_bear_top + eff_fvg_buf)
        // ── 4. За верхнюю границу Order Block медвежьего ──────────
        if not na(nearest_ob_bear) and nearest_ob_bear > entry
            array.push(sl_candidates, nearest_ob_bear + eff_ob_buf)
        // ── 5. За Fib 38.2% ───────────
        if not na(af_382) and af_382 > entry
            array.push(sl_candidates, af_382 + atr_val * 0.4)
        // ── 6. За верхнюю границу HVN зоны ─────────
        if not na(hvn1_price) and hvn1_price > entry
            array.push(sl_candidates, hvn1_price + hvn_zone_size + atr_val * 0.2)
        // ── Выбор лучшего SL кандидата ─────────
        if array.size(sl_candidates) > 0
            float best_sl = na
            float best_distance = na
            for i = 0 to array.size(sl_candidates) - 1
                float candidate = array.get(sl_candidates, i)
                float distance  = candidate - entry
                bool not_too_close = distance >= min_sl_dist
                bool not_too_far   = distance <= atr_fallback * 4
                if not_too_close and not_too_far
                    if na(best_sl) or distance < best_distance
                        best_sl       := candidate
                        best_distance := distance
            if na(best_sl)
                float sw_high_fb = use_struct_sl ? _sw_high_all : current_swing_high
                float buf = sw_high_fb * (sl_buffer_pct / 100.0)
                float raw_sl = sw_high_fb + buf
                sl := (raw_sl - entry) >= min_sl_dist ? raw_sl : entry + min_sl_dist
            else
                sl := best_sl
        else
            float sw_high_fb = use_struct_sl ? _sw_high_all : current_swing_high
            float buf = sw_high_fb * (sl_buffer_pct / 100.0)
            float raw_sl = sw_high_fb + buf
            sl := (raw_sl - entry) >= min_sl_dist ? raw_sl : entry + min_sl_dist
    // ✅ FIX: Абсолютный fallback — sl НИКОГДА не должен быть na
    if na(sl)
        sl := is_long ? entry - nz(atr_fallback, entry * 0.015) : entry + nz(atr_fallback, entry * 0.015)
    sl
